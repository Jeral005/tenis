function chamba(array, tipo){
  let html = "<ul>";
  for(let i=0; i<array.length; i++){
    if(tipo==="img"){
      html += "<li><img src='"+array[i]+"' alt='Imagen "+(i+1)+"'width='200'></li>";
    } else if(tipo==="pdf"){
      html += "<li><a href='"+array[i]+"' target='_blank'>PDF "+(i+1)+"</a></li>";
    } else if(tipo==="zip"){
      html += "<li><a href='"+array[i]+"' download>ZIP "+(i+1)+"</a></li>";
    } else if(tipo==="link"){
      html += "<li><a href='"+array[i]+"' target='_blank'>Link "+(i+1)+"</a></li>";
    }
  }
  html += "</ul>";
  return html;
}
